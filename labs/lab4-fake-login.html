async function authenticateUser(email, password, otpCode) {
    const user = await validateCredentials(email, password);
    
    if (!user) {
        throw new Error('Invalid credentials');
    }
    
    // يتطلب دائماً مصادقة ثنائية
    const validOTP = await verifyOTP(user.id, otpCode);
    if (!validOTP) {
        throw new Error('Invalid 2FA code');

// التحقق من الرابط من جانب العميل
function verifyLoginPage() {
    const legitDomain = 'techcorp.com';
    const currentDomain = window.location.hostname;
    
    // التحقق من النطاق الشرعي
    if (!currentDomain.endsWith(legitDomain)) {
        showWarning('Suspicious URL detected!');
        return false;
    }
    
    // التحقق من شهادة SSL
    if (window.location.protocol !== 'https:') {
        showWarning('Insecure connection!');
        return false;
    }
    
    // التحقق من المؤشرات البصرية
    displaySecurityBadge();
    return true;
}

📢 الاستجابة للحوادث

  1. أبلغ فريق الأمان عن الرسائل المشبوهة
  2. لا تنقر على الروابط أو تحمل المرفقات
  3. أعد توجيه رسائل التصيد الاحتيالي إلى: phishing@techcorp.com
  4. إذا تم إدخال بيانات الاعتماد، غيّر كلمة المرور فوراً
  5. فعّل المصادقة الثنائية على جميع الحسابات
  6. راقب الحسابات للأنشطة المشبوهة
